1. Literal
    literal: INTEGER
           | DOUBLE
           | STRING
           | BOOLEAN

2. Primary
    primary: varExpression
           | literal

3. Keyword
keyword: DEF
       | WHILE
       | FOR
       | IF
       | ELIF
       | NEW
       | AND
       | OR
       | RETURN
       | PRINT
       | VAR

4.
binaryOperator: PLUS
        | MINUS
        | comparator
        | MULT
        | MULTMULT

5. 
unaryOperator: PLUSPLUS
             | MINUSMINUS

6.
comparator: GREATER
          | LESS
          | GREATEQUAL
          | LESSEQUAL

7.
primary: literal
       | variable

8.
expression: primary
          | primary operator expression
          | primary unaryOperator
          | RETURN primary

9. statement
    statement: expression SEMI
             | RETURN primary SEMI


10.
conditional: primary comparator primary

11.
ifExpression: IF OPAREN conditional CPAREN body

12.
ifChain: ifExpression 
       | if Expression elifChain

13.
elifChain: ELSE body 
         | elifExpression
         | elifExpression elifChain 

14.
statement: functionDef
         | functionCall
         | expression SEMI
         | vardef

15. variable
    variable: functionCall
            | lambda
            | literal
            | expression

16. functionCall
    functionCall: varName OPAREN paramList CPAREN

17. paramList
    paramList: null
             | primary
             | primary paramList

18.
vardef: VAR varName variable

19.
comment: :) expression (:

20.
whileLoop: WHILE OPAREN conditional CPAREN body

21.
for: FOR OPAREN vardef

22.
expressionList: expression
              | expression optExpressionList

23.
body: OBRACKET expressionList CBRACKET

24.
functionDef: DEF VAR OPAREN paramDecList CPAREN body

25. lambda
    lambda: LAMBDA paramList body

26. variable definition
    varDefinition: VARDEF VAR EQUAL expression SEMI

27. lambda call
    lambdaCall: LAMBDA OPAREN paramList CPAREN

28. varExpression
    varExpression: VAR
                 | VAR OPAREN paramList CPAREN
29. paramDecList
    paramDecList: null
                | paramDec
                | paramDecList

30. paramDec
    paramDec: VARDEF VAR

31. statementList
    statementList: null
                 | statement
                 | statementList

32. elifExpression
    elifExpression: ELIF OPAREN constatement
